Classes¶

stdout_utf16¶

class stdout_utf16
{
public:
    stdout_utf16()
    {
        // Change stdout to Unicode. 
        // Cyrillic and Ideographic characters will appear in the console (console is unicode).
        _setmode(_fileno(stdout), _O_U16TEXT);
    }

    ~stdout_utf16()
    {
        // restore ANSI mode
        _setmode(_fileno(stdout), _O_TEXT); 
    }
};
